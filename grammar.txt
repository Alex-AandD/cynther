program: declaration* EOF
declaration: vardeclaration
	    |stmt

stmt:= expressionstmt 
      |if expr block
      |while expr block 
      |block

expr := binary
       |unary
       |primary

binary:= comparison
	|equality
	|term
	|factor

comparison:= equality (> | < | <= | >=) comparison*
equality:= and ( (== | !=) and) *
and:= or (&& or)*
or := term (|| term)*
term:= factor ((+ | -) factor)*
factor:= unary ((* | /) unary)*
unary:= ((! | -) unary)* | primary
primary:= bool | int | double | string | ( expression )
